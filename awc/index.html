An HTTP Client

use actix_rt::System;
use awc::Client;

#[actix_rt::main]
async fn main() {
   let mut client = Client::default();

   let response = client.get("http://www.rust-lang.org") // <- Create request builder
       .header("User-Agent", "Actix-web")
       .send()                             // <- Send http request
       .await;

    println!("Response: {:?}", response);
}
